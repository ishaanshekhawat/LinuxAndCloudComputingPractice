// Create directories `lab1/lab2/lab3` using one command.

mkdir lab{1..3}

// Create 10 files (`file1` to `file10`) inside `lab3` using one command.

touch file{1..10}

// Display who is logged in, and the current user.

who
whoami

// List your processes and terminate one.

ps                 # To check all the processes
sleep 30 &         # To create a process in the bg that tells the shell to go to sleep for 30 seconds.
ps aux             # To check the prcoesses for all users, in user-oriented format, including processes that are not attached to terminal.
kill -9 20708      # Signal a kill command for the process number 20708 (sleep process)
ps aux             # To confirm that the process with number 20708 is deleted

// Create 50 files `data1` to `data50` inside a directory with a single command.

touch data{1..50}

// Copy all `.txt` files from one folder to another using wildcards.

cp /folder1/folder2/*.txt .            # Copies all the txt files inside /folder1/folder2 to current directory

//  View file contents.

cat file1             
head -n 20 file1        # Display the first 20 lines in the file
tail -n 20 file1        # Display the last 20 lines in the file
less file1              # Provides a scrollable viewer
more file1              # Displays one screen at a time
